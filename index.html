今日学习总结 

/*筛选数组
filter()方法创建一个新的数组，新数组中的元素是通过检查制定数组中符合条件的所有元素，
主要用于筛选数组。
*/
var arr = [12, 52, 5, 40];
var newArr = arr.filter(function(value, index) {
    // return value > 20;
    return value % 2 == 0;
});
console.log(newArr);

/*迭代数组
forEach()
array.forEach(function(currentValue, index, arr) {
    *************
    *************
})
currentValue: 数组当前的值
index: 数组当前的索引
arr: 数组对象本身
*/
var arr = [12, 52, 5, 40];
var sum = 0;
arr.forEach(function(value, index, array) {
    console.log('每个数组元素' + value);
    console.log('每个数组索引号' + index);
    console.log('每个数组本身' + array);
    sum += value;
}) 
console.log(sum);

/*some()
array.some(function(currentValue, index, arr) {
    ******************
    ******************
})
some()方法用于检测数组中的元素是否满足指定条件，通俗点就是查找数组中是否有满足条件的元素。
注意它返回的是布尔值，如果查找到这个元素，就返回true，查不到就返回false。
*/
var arr = [12, 52, 5, 40];
var dge = arr.some(function(value){
    // return value >= 50;
    return value > 100; 
})
console.log(dge)
